[docs]    def run(self):
        LOGGER.info(f"Cleanup checkpoints in {self.path_model}/{self.path_checkpoints}")
        checkpoint_files = Path(self.path_model, self.path_checkpoints).glob("model.ckpt-*")
        for checkpoint_file in checkpoint_files:
